Codebase performance optimizations

Two core R script files were refactored to optimize performance, memory safety, and code readability.

Distinction calculation (Functions/distinction.R)

  1. Pre-allocated vectors:
    • Issue: d <- rep(n, 0) and u <- rep(n, 0) created zero-length vectors. Dynamically growing vectors inside a for loop causes expensive memory reallocation on every iteration.
    • Fix: Pre-allocated the vectors to full length using rep(0, n) (or numeric(n)).
  2. Logical NULL-checking:
    • Issue: Coerced logical values to integers (is.null(V(x)$name) == 1). If V(x)$name was originally NULL, both the if and subsequent if blocks executed redundantly.
    • Fix: Simplified to standard if (is.null(V(x)$name)) { ... } else { ... }.

The optimized function is defined below:

distinction_opt <- function(x, norm = "abm", digits = 4) {
  require(igraph)
  n <- vcount(x)
  if (is.null(V(x)$name)) {
    V(x)$name <- 1:n
    names <- V(x)$name
  } else {
    names <- V(x)$name
    V(x)$name <- 1:n
  }
  
  norm.max <- function(x) { x^2 / max(x^2) }
  norm.one <- function(x) { x^2 }
  norm.abm <- function(x) { abs(x) / max(abs(x)) }
  norm.abs <- function(x) { abs(x) }
  
  eig.x <- eigen(as_adjacency_matrix(x))
  xc <- components(x)
  nc <- xc$no
  nn <- sum(xc$csize != 1)
  ni <- nc - nn
  ci <- which(xc$csize != 1)
  
  if (nc == 1) { s <- eig.x$vectors[, 1] }
  if (nc > 1 & nn == 1) { s <- eig.x$vectors[, 1] }
  if (nc > 1 & ni == 1) { s <- eig.x$vectors[, 1] }
  if (nc > 1 & nn > 1) { s <- rowSums(eig.x$vectors[, ci]) }
  
  if (norm == "max") { s <- norm.max(s) }
  if (norm == "one") { s <- norm.one(s) }
  if (norm == "abm") { s <- norm.abm(s) }
  if (norm == "abs") { s <- norm.abs(s) }
  
  # Correctly pre-allocated vectors
  d <- rep(0, n)
  u <- rep(0, n)
  
  for (i in 1:n) {
    j <- neighbors(x, i)
    xd <- delete_vertices(x, i)
    ed <- ecount(xd) == 0
    nd <- vcount(xd)
    ncd <- components(xd)$no
    eig.xd <- eigen(as_adjacency_matrix(xd))
    xdc <- components(xd)
    nnd <- sum(xdc$csize != 1)
    nid <- ncd - nnd
    cid <- which(components(xd)$csize != 1)
    
    if (ncd == 1) { sd <- eig.xd$vectors[, 1] }
    if (ncd > 1 & nnd == 1) { sd <- eig.xd$vectors[, 1] }
    if (ncd > 1 & nid == 1) { sd <- eig.xd$vectors[, 1] }
    if (ncd > 1 & nnd > 1) { sd <- rowSums(eig.xd$vectors[, cid]) }
    
    if (ed == 0 & nd > 1 & norm == "max") { sd <- norm.max(sd) }
    if (ed == 0 & nd > 1 & norm == "one") { sd <- norm.one(sd) }
    if (ed == 0 & nd > 1 & norm == "abm") { sd <- norm.abm(sd) }
    if (ed == 0 & nd > 1 & norm == "abs") { sd <- norm.abs(sd) }
    if (ed == 1 & nd > 1 | nd == 1) { sd <- rep(0, nd) }
    
    names(sd) <- V(xd)$name
    u[i] <- mean(sd[as.character(j)])
    d[i] <- s[i] - u[i]
  }
  
  d[is.na(d)] <- 0
  u[is.na(u)] <- 0
  scalar <- mean(u) / mean(s)
  scd <- (s * scalar) - u
  dat <- data.frame(d = d, scd = scd, s = s, u = u, scalar = scalar)
  dat <- round(dat, digits)
  dat <- data.frame(n = names, dat)
  rownames(dat) <- 1:nrow(dat)
  return(dat)
}
